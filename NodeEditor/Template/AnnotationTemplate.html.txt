    <script>
        // 获取所有目录项和内容区域
        const tocItems = document.querySelectorAll('.toc-item');
        const contentSections = document.querySelectorAll('.content-section');
        const contentSectionNodes = document.querySelectorAll('.content-section-node');

        contentSectionNodes.forEach(node => node.classList.add('active'));

        // 开关
        const toggleSwitch = document.getElementById('toggleSwitch');
        const statusText = document.getElementById('statusText');

        toggleSwitch.addEventListener('change', function() {
            if (this.checked) {
                contentSectionNodes.forEach(node => {
                    const dataType = node.dataset.type ?? '';
                    if (dataType.indexOf("不常用") != -1 || dataType.indexOf("不显示") != -1) {
                        node.classList.remove('active');
                    }
                    else {
                        node.classList.add('active');
                    }
                });
                statusText.textContent = '常用';
                statusText.style.color = '#2196F3';
            } else {
                contentSectionNodes.forEach(node => node.classList.add('active'));
                statusText.textContent = '全部';
                statusText.style.color = '#666';
            }
        });

        // 为每个目录项添加点击事件
        tocItems.forEach(item => {
            item.addEventListener('click', () => {
                const targetId = item.dataset.target;
                
                // 移除所有激活状态
                tocItems.forEach(i => i.classList.remove('active'));
                contentSections.forEach(section => section.classList.remove('active'));

                // 添加当前激活状态
                item.classList.add('active');
                document.getElementById(targetId).classList.add('active');
            });
        });

        // 处理文件夹收起
        document.querySelectorAll('.directory li').forEach(item => {
            if (item.querySelector('ul')) {
                item.classList.add('collapsed');
                item.addEventListener('click', function(e) {
                    console.log(e.target.tagName);
                    if (e.target.tagName == 'DIV') {
                        // this.classList.toggle('avtive');
                    }
                    else {
                        // 收缩
                        this.classList.toggle('collapsed');
                    }
                });
            }
        });
    </script>
